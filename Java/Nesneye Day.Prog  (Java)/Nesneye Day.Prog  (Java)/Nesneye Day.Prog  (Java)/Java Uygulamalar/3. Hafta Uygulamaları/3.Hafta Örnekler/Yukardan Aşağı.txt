import java.util.Scanner;


public class deneme1 {

	public static void main(String[] args) {
		
		Scanner klavye=new Scanner(System.in);
		System.out.println("kaç satır?:");
		int satırsayısı=klavye.nextInt();
		int bosluksayısı=0;
		int yıldızsayısı=satırsayısı*2-1;
		//kullanıcının girdiği satır sayısının iki katının
		//her bir eksiği yıldız sayısı olarak belirlendi.
		for (int i = 0; i < satırsayısı; i++) {
			for (int j = 0; j < bosluksayısı; j++) {
				System.out.print(" ");
			}
			for (int j = 0; j < yıldızsayısı; j++) {
				//Bu DÖngü yıldızı basmaya yarıyor...
				System.out.print("*");
			}
			System.out.println();
			bosluksayısı++;
			yıldızsayısı-=2;
		}	
	}
}
